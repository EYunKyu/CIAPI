Examples
다음은 CCamera 클래스 사용에 대한 예시입니다.
int nCameraPick = Define.MATROX;
CCamera m_Camera = CCamera.CameraPick(nCameraPick, m_nCamNumber, RecipePath + "\\");

m_Camera.m_dExposureTime = theRecipe.m_dExposureTime[m_nCamNumber];
m_Camera.m_dGain = theRecipe.m_dGain[m_nCamNumber];
m_Camera.m_sCurrentEquipment = theRecipe.m_sCurrentEquipment;

m_Camera.CamActionInfo = CameraStatus;                           
m_Camera.Initialize(m_Camera);
m_Camera.Connect();

m_bIsConnected = m_Camera.IsConnected();
